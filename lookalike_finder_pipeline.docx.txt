Обзор пайплайна
Пайплайн автоматически находит компании — потенциальных клиентов Interexy — на основе портфолио-кейсов. Для каждого кейса система строит детальный технический профиль, генерирует поисковые запросы, парсит сайты найденных компаний, сравнивает их с профилем и расставляет тиры близости. Tier 1–2 получают углублённый ресёрч с поиском decision makers.

Шаг 1 — Анализ кейса Interexy
1
	Case Analysis
Читаем портфолио-страницу кейса, извлекаем технический профиль через OpenAI
	

На входе — URL кейса на сайте Interexy (или текст вручную). Jina Reader извлекает чистый текст страницы, затем OpenAI структурирует его в JSON-профиль.
1.1 Извлечение текста кейса
python
import requests, os
from openai import OpenAI
 
JINA_KEY   = os.environ["JINA_API_KEY"]
OPENAI_KEY = os.environ["OPENAI_API_KEY"]
client = OpenAI(api_key=OPENAI_KEY)
 
def fetch_case_text(case_url: str) -> str:
    """Читает страницу кейса через Jina Reader."""
    headers = {
        "Authorization": f"Bearer {JINA_KEY}",
        "Accept": "application/json",
        "X-Return-Format": "markdown"
    }
    r = requests.get(f"https://r.jina.ai/{case_url}", headers=headers)
    r.raise_for_status()
    return r.json()["data"]["content"]

Шаг 2 — Профиль компании-клиента
2
	Client Company Profile
Serper собирает публичную информацию о компании из кейса, LLM строит полный профиль
	

Зная имя компании из шага 1, делаем несколько Serper-запросов: funding, новости, LinkedIn, стек. LLM синтезирует в единый профиль потенциального клиента.
queries генерируется gpt-4o для нескольких запросов по базовой компании. (как в шаге 3), но только для базовой компании по которой формируется профиль.
2.1 Serper — сбор информации о компании
python
SERPER_KEY = os.environ["SERPER_KEY"]
 
def serper_search(query: str, num: int = 5) -> list[dict]:
    r = requests.post(
        "https://google.serper.dev/search",
        headers={"X-API-KEY": SERPER_KEY, "Content-Type": "application/json"},
        json={"q": query, "num": num, "gl": "us", "hl": "en"}
    )
    r.raise_for_status()
    return r.json().get("organic", [])
 
def collect_company_intel(company_name: str) -> list[dict]:
    """Собирает разностороннюю информацию о компании через Serper."""
    queries = [
        f"{company_name} funding investors",
        f"{company_name} product features technology stack",
        f"{company_name} team CEO CTO founder",
        f"{company_name} news 2024 2025",
        f"site:linkedin.com {company_name}",
    ]
    all_results = []
    for q in queries:
        hits = serper_search(q, num=3)
        for h in hits:
            all_results.append({
                "query": q,
                "title": h.get("title", ""),
                "snippet": h.get("snippet", ""),
                "url": h.get("link", ""),
            })
    return all_results

def build_client_profile(company_name: str, search_results: list) -> dict:
    results_text = "\n".join(
        f"[{r['query']}] {r['title']}: {r['snippet']}"
        for r in search_results
    )
    resp = client.chat.completions.create(
        model="gpt-4o",
        temperature=0,
        messages=[
            {"role": "system", "content": CLIENT_PROFILE_SYSTEM},
            {"role": "user", "content": CLIENT_PROFILE_USER.format(
                company_name=company_name,
                search_results=results_text[:6000]
            )}
        ]
    )
    import json
    return json.loads(resp.choices[0].message.content)

Шаг 4 — Поиск и парсинг сайтов
4
	Search & Parse
Serper находит компании по запросам, Jina Reader читает их сайты
	

По каждому запросу Serper возвращает топ-5 результатов. Дедупликация по URL. Jina Reader параллельно читает сайты всех найденных компаний и извлекает чистый текст для анализа.
4.1 Сбор компаний через Serper
python
from concurrent.futures import ThreadPoolExecutor
from urllib.parse import urlparse
 
def search_companies(queries: list[str]) -> list[dict]:
    """Ищет компании по всем запросам, дедуплицирует по домену."""
    all_results = []
    for q in queries:
        hits = serper_search(q, num=5)
        for h in hits:
            all_results.append({
                "url":     h.get("link", ""),
                "title":   h.get("title", ""),
                "snippet": h.get("snippet", ""),
                "query":   q,
            })
 
    # Дедупликация по корневому домену
    seen_domains = set()
    unique = []
    for r in all_results:
        domain = urlparse(r["url"]).netloc.replace("www.", "")
        if domain and domain not in seen_domains:
            seen_domains.add(domain)
            r["domain"] = domain
            unique.append(r)
    return unique


4.2 Параллельный парсинг через Jina Reader
python
def jina_read(url: str) -> str:
    """Читает сайт компании через Jina Reader, возвращает Markdown."""
    try:
        headers = {
            "Authorization": f"Bearer {JINA_KEY}",
            "Accept": "application/json",
            "X-Return-Format": "markdown",
            "X-Timeout": "15"
        }
        r = requests.get(f"https://r.jina.ai/{url}", headers=headers, timeout=20)
        if r.status_code == 200:
            data = r.json()
            return data.get("data", {}).get("content", "")[:5000]
        return ""
    except Exception:
        return ""
 
def parse_all_companies(companies: list[dict]) -> list[dict]:
    """Параллельно парсит сайты всех найденных компаний."""
    def enrich(company):
        company["page_content"] = jina_read(company["url"])
        return company
 
    with ThreadPoolExecutor(max_workers=8) as ex:
        return list(ex.map(enrich, companies))


4.3 Фильтрация нерелевантных результатов
До тиеринга отфильтровываем явный мусор: новостные агрегаторы, маркетплейсы разработчиков, сами dev-агентства. Это делает LLM быстрой pre-filter проверкой на gpt-4o-mini (дёшево).
python
PREFILTER_PROMPT = """
Ответь ТОЛЬКО "yes" или "no".
Является ли эта страница сайтом реального продуктового стартапа или компании,
которая строит свой продукт (не dev-агентства, не новостные сайты,
не маркетплейсы разработчиков)?
 
URL: {url}
Заголовок: {title}
Сниппет: {snippet}
Контент (первые 500 символов): {content}
"""
 
def is_relevant_company(company: dict) -> bool:
    resp = client.chat.completions.create(
        model="gpt-4o-mini",
        temperature=0,
        messages=[{"role": "user", "content": PREFILTER_PROMPT.format(
            url=company["url"],
            title=company["title"],
            snippet=company["snippet"],
            content=company.get("page_content", "")[:500]
        )}]
    )
    return resp.choices[0].message.content.strip().lower() == "yes"
Шаг 5 — Тиеринг компаний
5
	Tiering
LLM сравнивает каждую компанию с профилем кейса и присваивает тир
	

5.1 Определение тиров
Тир
	Название
	Критерий
	T1
	Горячий лид
	Тот же тип продукта + явные сигналы найма разработчика (вакансии, недавний раунд, нет tech co-founder)
	T2
	Тёплый лид
	Тот же тип продукта, нет явных сигналов найма — но профиль компании совпадает
	T3
	Похожая вертикаль
	Смежный продукт в той же отрасли — другая технология или другой сегмент рынка
	T4
	Слабое совпадение
	Другая вертикаль, но подходит под экспертизу Interexy (IoT, mobile, compliance)
	—
	Нерелевантен
	Продукт не совпадает или это dev-агентство / новостной сайт

Профиль кейса Interexy (эталон для сравнения):
{case_profile}

URL: {url}
Название: {title}

Определи тир и верни JSON:
{
  "company_name": "название компании",
  "website": "url",
  "tier": "T1" | "T2" | "T3" | "T4" | "none",
  "tier_reason": "2-3 предложения почему именно этот тир",
  "product_match": "краткое описание совпадения продукта",
  "outsource_signals": ["список сигналов что ищут разработчика"],
  "tech_overlap": ["пересечение технологий"],
  "vertical": "вертикаль компании",
  "region": "регион компании",
  "stage": "стадия компании",
  "confidence": 0.0-1.0
}
"""
 
def tier_company(company: dict, case_profile: dict) -> dict:
    import json
    resp = client.chat.completions.create(
        model="gpt-4o",
        temperature=0,
        messages=[
            {"role": "system", "content": TIERING_SYSTEM},
            {"role": "user",   "content": TIERING_USER.format(
                case_profile=json.dumps(case_profile, ensure_ascii=False),
                url=company["url"],
                title=company["title"],
                snippet=company["snippet"],
                page_content=company.get("page_content", "")[:3000]
            )}
        ]
    )
    result = json.loads(resp.choices[0].message.content)
    result["source_query"] = company.get("query", "")
    return result
 
def tier_all_companies(companies: list[dict], case_profile: dict) -> dict:
    """Тирирует все компании, возвращает словарь {T1: [...], T2: [...], ...}"""
    tiered = {"T1": [], "T2": [], "T3": [], "T4": [], "none": []}
 
    with ThreadPoolExecutor(max_workers=5) as ex:
        results = list(ex.map(
            lambda c: tier_company(c, case_profile),
            companies
        ))
 
    for r in results:
        tier = r.get("tier", "none")
        tiered.get(tier, tiered["none"]).append(r)
 
    # Сортируем внутри каждого тира по confidence
    for tier in tiered:
        tiered[tier].sort(key=lambda x: x.get("confidence", 0), reverse=True)
 
    return tiered
Шаг 6 — Deep Research для Tier 1–2
6
	Deep Research
Для горячих и тёплых лидов: новости, funding, decision makers через Apollo
	

Компании Tier 1 и Tier 2 получают углублённый ресёрч. Для каждой компании: дополнительный Serper-поиск (funding, вакансии, новости), затем Apollo — поиск CTO / VP Engineering / Head of Product.
6.1 Углублённый Serper-ресёрч
python
def deep_research_company(company: dict) -> dict:
    """Собирает дополнительную информацию для Tier 1-2 компании."""
    name = company.get("company_name", "")
    domain = company.get("website", "").replace("https://", "").replace("www.", "").split("/")[0]
 
    research_queries = [
        f'"{name}" funding round 2024 2025',
        f'"{name}" hiring CTO "VP Engineering" OR "Head of Product"',
        f'"{name}" crunchbase',
        f'site:linkedin.com/in "{name}"',
    ]
 
    snippets = []
    for q in research_queries:
        hits = serper_search(q, num=3)
        for h in hits:
            snippets.append(f"{h.get('title','')}: {h.get('snippet','')}")
 
    company["deep_research_snippets"] = snippets
    return company

APOLLO_KEY = os.environ["APOLLO_API_KEY"]
 
def find_decision_makers(domain: str) -> list[dict]:
    """Ищет CTO / VP Engineering / Head of Product через Apollo."""
    url = "https://api.apollo.io/api/v1/mixed_people/search"
    headers = {
        "x-api-key": APOLLO_KEY,
        "Content-Type": "application/json",
        "accept": "application/json",
    }
    payload = {
        "organization_domains": [domain],
        "person_titles": [
            "CTO",
            "Chief Technology Officer",
            "VP Engineering",
            "VP of Engineering",
            "Head of Product",
            "VP Product",
            "Co-Founder",
            "Founder",
        ],
        "page": 1,
        "per_page": 5,
    }
    try:
        r = requests.post(url, json=payload, headers=headers)
        r.raise_for_status()
        people = r.json().get("people", [])
        return [{
            "name":     p.get("name"),
            "title":    p.get("title"),
            "email":    p.get("email"),

def enrich_tier1_tier2(tiered: dict) -> dict:

            company["decision_makers"] = find_decision_makers(domain)
    return tiered

7.1 Генерация value proposition

VP_SYSTEM = """
Ты — B2B sales copywriter.
Пишешь первое outreach-сообщение от аутсорс-компании Interexy.
Тон: профессиональный, конкретный, без продажных клише.
Язык: английский.
Длина: 3-4 предложения.
"""
 
VP_USER = """
Наш кейс (что мы построили):
{case_summary}
 
Компания получателя:
Название: {company_name}
Продукт: {product_match}
Контакт: {contact_name}, {contact_title}
 
Напиши 3-4 предложения для первого outreach.
Упомяни конкретику: наш кейс + их продукт.
Не используй "I hope this email finds you well" и подобное.
"""
 
def generate_value_prop(company: dict, case_profile: dict) -> str:
    contact = (company.get("decision_makers") or [{}])[0]
    case_summary = (
        f"Мы построили {case_profile['product_description']} "
        f"для {case_profile['client_company']}. "
        f"Стек: {', '.join(case_profile['tech_stack'][:4])}."
    )
    resp = client.chat.completions.create(
        model="gpt-4o",
        temperature=0.4,
        messages=[
            {"role": "system", "content": VP_SYSTEM},
            {"role": "user",   "content": VP_USER.format(
                case_summary=case_summary,
                company_name=company.get("company_name", ""),
                product_match=company.get("product_match", ""),
                contact_name=contact.get("name", "there"),
                contact_title=contact.get("title", ""),
            )}
        ]
    )
    return resp.choices[0].message.content.strip()

import json
from datetime import date
 
def build_final_output(tiered: dict, case_profile: dict) -> dict:
    """Добавляет value prop для T1-T2, формирует итоговую структуру."""
    for tier in ["T1", "T2"]:
        for company in tiered[tier]:
            company["value_proposition"] = generate_value_prop(company, case_profile)
 
    total = sum(len(v) for v in tiered.values())
    return {
        "case_name":    case_profile["case_name"],
        "run_date":     str(date.today()),
        "total_found":  total,
        "tiers":        tiered,
    }
 
def save_output(output: dict, path: str = None):
    filename = path or f"output_{output['case_name']}_{output['run_date']}.json"
    with open(filename, "w", encoding="utf-8") as f:
        json.dump(output, f, ensure_ascii=False, indent=2)
    print(f"Saved: {filename}")
    print(f"T1: {len(output['tiers']['T1'])} | T2: {len(output['tiers']['T2'])} | "
          f"T3: {len(output['tiers']['T3'])} | T4: {len(output['tiers']['T4'])}")
